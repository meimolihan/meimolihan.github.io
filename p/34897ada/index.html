log_message() {

# ÂáΩÊï∞ÔºöÊ£ÄÊü• NFS ÊúçÂä°Âô®
check_nfs_server() {
    local server_ip=$1
    local server_name=$2
    
    echo -e "\n${BLUE}Ê£ÄÊü•ÊúçÂä°Âô®: $server_name ($server_ip)${NC}" | tee -a "$LOG_FILE"
    
    # Ê£ÄÊü•ÊúçÂä°Âô®ÊòØÂê¶Âú®Á∫ø
    if ping -c 1 -W 1 "$server_ip" &> /dev/null; then

        # Ê£ÄÊü• NFS ÊúçÂä°

            # Ëé∑ÂèñÂÖ±‰∫´ÂàóË°®

            if [ $? -eq 0 ]; then

                    echo -e "  ${GREEN}$line${NC}" | tee -a "$LOG_FILE"
                done
                
                # Ê£ÄÊü•ÊåÇËΩΩÁÇπ
                check_nfs_mounts "$server_ip"
            else
                echo -e "${YELLOW}‚ö† Êó†Ê≥ïËé∑ÂèñÂÖ±‰∫´ÂàóË°®${NC}" | tee -a "$LOG_FILE"
            fi
        else
            echo -e "${RED}‚úó NFS ÊúçÂä°ÂºÇÂ∏∏${NC}" | tee -a "$LOG_FILE"

        fi

        echo -e "${RED}‚úó ÊúçÂä°Âô®Á¶ªÁ∫ø${NC}" | tee -a "$LOG_FILE"

}

# ÂáΩÊï∞ÔºöÊ£ÄÊü• NFS ÊåÇËΩΩÁÇπ
check_nfs_mounts() {
    local server_ip=$1
    mount | grep "nfs" | grep "$server_ip" | while read -r line; do
        local mount_point
        mount_point=$(echo "$line" | awk '{print $3}')
        local device
        device=$(echo "$line" | awk '{print $1}')
        
        # Ê£ÄÊü•ÊåÇËΩΩÁÇπÊòØÂê¶ÂèØËÆøÈóÆ
        if touch "$mount_point/.nfs_test" 2>/dev/null; then
            echo -e "  ${GREEN}‚úì ÊåÇËΩΩÁÇπ $mount_point Ê≠£Â∏∏${NC}" | tee -a "$LOG_FILE"
            rm -f "$mount_point/.nfs_test"
        else
            echo -e "  ${RED}‚úó ÊåÇËΩΩÁÇπ $mount_point ÂºÇÂ∏∏${NC}" | tee -a "$LOG_FILE"
            log_message "ERROR: Mount point $mount_point ($device) is not accessible"
            
            # Â∞ùËØïÈáçÊñ∞ÊåÇËΩΩ
            umount -f "$mount_point" 2>/dev/null
            mount "$device" "$mount_point" 2>/dev/null

                echo -e "  ${GREEN}‚úì ÊåÇËΩΩÁÇπ $mount_point Â∑≤ÊÅ¢Â§ç${NC}" | tee -a "$LOG_FILE"
                log_message "INFO: Mount point $mount_point recovered"
            fi
        fi
    done
}

main() {
    echo "=== NFS ÂÖ±‰∫´Áä∂ÊÄÅÊ£ÄÊü• ===" | tee -a "$LOG_FILE"
    echo "ÂºÄÂßãÊó∂Èó¥: $(date)" | tee -a "$LOG_FILE"

        check_nfs_server "$server_ip" "$server_name"
    done
    
    # Ê£ÄÊü•Êú¨Âú∞ NFS ÊúçÂä°Áä∂ÊÄÅ
    echo -e "\n${BLUE}Ê£ÄÊü•Êú¨Âú∞ NFS ÊúçÂä°Áä∂ÊÄÅ${NC}" | tee -a "$LOG_FILE"
    systemctl status nfs-kernel-server --no-pager | tee -a "$LOG_FILE"
    
    echo -e "\nÊ£ÄÊü•ÂÆåÊàêÊó∂Èó¥: $(date)" | tee -a "$LOG_FILE"
    echo "ËØ¶ÁªÜÊó•ÂøóËØ∑Êü•Áúã: $LOG_FILE" | tee -a "$LOG_FILE"
}

# ÊâßË°å‰∏ªÂáΩÊï∞
main "$@"
EOF
