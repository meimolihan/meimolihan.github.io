#!/bin/bash
apt update && apt install samba -y
mkdir -pm 755 /mnt && cat >>/etc/samba/smb.conf <<'MYENDMARK'
########   жҲ‘зҡ„sambaе…ұдә«и®ҫзҪ®   ########
[Debian]
     ## иҝҷжҳҜжҲ‘зҡ„е…ұдә«еҗҚз§°
     comment = mysamba
     ## е…ұдә«зӣ®еҪ•зҡ„и·Ҝеҫ„
     path = /mnt
     ## жҢҮе®ҡе…ұдә«жҳҜеҗҰеә”иҜҘеңЁзҪ‘з»ңйӮ»еұ…дёӯиў«жөҸи§ҲеҲ°пјҢyesжҳҫзӨәе…ұдә«еҗҚз§°пјҢnoйҡҗи—Ҹе…ұдә«еҗҚз§°гҖӮ
     browseable = yes
     ## жҳҜеҗҰе…Ғи®ёз”ЁжҲ·еҶҷе…ҘжӯӨе…ұдә«пјҢyesдёәеҸҜеҶҷе…ҘпјҢnoдёәдёҚеҸҜеҶҷе…ҘгҖӮ
     writeable = yes
     ## жҢҮе®ҡе…ұдә«з”ЁжҲ·жҳҜеҗҰеҸҜиҜ»еҶҷпјҢyesдёәеҸӘиҜ»пјҢnoдёәиҜ»еҶҷгҖӮ
     read only = no
     ## ж–°е»әж–Үд»¶зҡ„й»ҳи®ӨжқғйҷҗжҺ©з Ғ
     create mask = 0777
     ## ж–°е»әзӣ®еҪ•зҡ„й»ҳи®ӨжқғйҷҗжҺ©з Ғ
     directory mask = 0777
     ## е…Ғи®ёи®ҝе®ўеҗҰ
     guest ok = no
     ## ејәеҲ¶з”ЁжҲ·дёәroot
     force user = root
     ## ејәеҲ¶з»„дёәroot
     force group = root
     ## е®ҡд№үз®ЎзҗҶе‘ҳз”ЁжҲ·еҲ—иЎЁ
     admin users = admin
     ## е®ҡд№үе…Ғи®ёи®ҝй—®жӯӨе…ұдә«зҡ„жңүж•Ҳз”ЁжҲ·еҲ—иЎЁпјҢд№ҹеҸҜд»ҘжҳҜз»„еҗҚпјҲд»Ҙ @ ејҖеӨҙпјү
     valid users = admin,root
     ## иҰҒжұӮеҜҶз Ғи®ҝй—®
     password required = yes
     ## е…ұдә«жҳҜеҗҰеҸҜз”ЁпјҢ yesдёәжҳҫзӨәе…ұдә«пјҢno дёәйҡҗи—Ҹе…ұдә«
     available = yes
     ## еҜ№дәҺAppleж–Үд»¶иҝӣиЎҢзј–з Ғ
     fruit:encoding = native
     ## еҜ№дәҺAppleж–Үд»¶е…ғж•°жҚ®
     fruit:metadata = stream
     ## и®ҫзҪ®дёә no иЎЁзӨәSambaдёҚдјҡжӢ’з»қAppleDoubleж–Үд»¶
     fruit:veto_appledouble = no
     ## иҝҷдәӣVFSжЁЎеқ—еҸҜд»ҘеўһејәSambaжңҚеҠЎеҷЁзҡ„еҠҹиғҪ
     vfs objects = catia fruit streams_xattr
MYENDMARK

## ж–°е»әadminз”ЁжҲ· && дёәadminз”ЁжҲ·еҲӣе»әдёӨж¬ЎеҜҶз Ғ
useradd admin && (echo иҫ“е…ҘsmbеҜҶз Ғ; echo иҫ“е…ҘsmbеҜҶз Ғ) | smbpasswd -a -s admin
systemctl restart smbd
EOF

#!/bin/bash
yum update -y && yum install samba -y
mkdir -pm 755 /mnt && cat >>/etc/samba/smb.conf <<'MYENDMARK'
[CentOS]
     ## иҝҷжҳҜжҲ‘зҡ„е…ұдә«еҗҚз§°
     comment = mysamba
     ## е…ұдә«зӣ®еҪ•зҡ„и·Ҝеҫ„
     path = /mnt
     ## жҢҮе®ҡе…ұдә«жҳҜеҗҰеә”иҜҘеңЁзҪ‘з»ңйӮ»еұ…дёӯиў«жөҸи§ҲеҲ°пјҢyesжҳҫзӨәе…ұдә«еҗҚз§°пјҢnoйҡҗи—Ҹе…ұдә«еҗҚз§°гҖӮ
     browseable = yes
     ## жҳҜеҗҰе…Ғи®ёз”ЁжҲ·еҶҷе…ҘжӯӨе…ұдә«пјҢyesдёәеҸҜеҶҷе…ҘпјҢnoдёәдёҚеҸҜеҶҷе…ҘгҖӮ
     writeable = yes
     ## жҢҮе®ҡе…ұдә«з”ЁжҲ·жҳҜеҗҰеҸҜиҜ»еҶҷпјҢyesдёәеҸӘиҜ»пјҢnoдёәиҜ»еҶҷгҖӮ
     read only = no
     ## ж–°е»әж–Үд»¶зҡ„й»ҳи®ӨжқғйҷҗжҺ©з Ғ
     create mask = 0777
     ## ж–°е»әзӣ®еҪ•зҡ„й»ҳи®ӨжқғйҷҗжҺ©з Ғ
     directory mask = 0777
     ## е…Ғи®ёи®ҝе®ўеҗҰ
     guest ok = no
     ## ејәеҲ¶з”ЁжҲ·дёәroot
     force user = root
     ## ејәеҲ¶з»„дёәroot
     force group = root
     ## е®ҡд№үз®ЎзҗҶе‘ҳз”ЁжҲ·еҲ—иЎЁ
     admin users = admin
     ## е®ҡд№үе…Ғи®ёи®ҝй—®жӯӨе…ұдә«зҡ„жңүж•Ҳз”ЁжҲ·еҲ—иЎЁпјҢд№ҹеҸҜд»ҘжҳҜз»„еҗҚпјҲд»Ҙ @ ејҖеӨҙпјү
     valid users = admin,root
     ## иҰҒжұӮеҜҶз Ғи®ҝй—®
     password required = yes
     ## е…ұдә«жҳҜеҗҰеҸҜз”ЁпјҢ yesдёәжҳҫзӨәе…ұдә«пјҢno дёәйҡҗи—Ҹе…ұдә«
     available = yes
     ## еҜ№дәҺAppleж–Үд»¶иҝӣиЎҢзј–з Ғ
     fruit:encoding = native
     ## еҜ№дәҺAppleж–Үд»¶е…ғж•°жҚ®
     fruit:metadata = stream
     ## и®ҫзҪ®дёә no иЎЁзӨәSambaдёҚдјҡжӢ’з»қAppleDoubleж–Үд»¶
     fruit:veto_appledouble = no
     ## иҝҷдәӣVFSжЁЎеқ—еҸҜд»ҘеўһејәSambaжңҚеҠЎеҷЁзҡ„еҠҹиғҪ
     vfs objects = catia fruit streams_xattr
MYENDMARK

## ж–°е»әadminз”ЁжҲ· && дёәadminз”ЁжҲ·еҲӣе»әдёӨж¬ЎеҜҶз Ғ
useradd admin && (echo иҫ“е…ҘsmbеҜҶз Ғ; echo иҫ“е…ҘsmbеҜҶз Ғ) | smbpasswd -a -s admin
systemctl restart smb
EOF
