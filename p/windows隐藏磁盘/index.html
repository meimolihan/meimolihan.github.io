一 、 参看磁盘详细信息

1 、Win10参看磁盘详细信息批处理，文本编辑保存为,bat批处理文件

@echo off
%1 mshta vbscript:CreateObject("Shell.Application").ShellExecute("cmd.exe","/c %~s0 ::","","runas",1)(window.close)&&exit
cd /d "%~dp0"
rem 上面是bat批处理以管理员权限运行
rem 下面是参看磁盘详细信息的命令
echo list volume > commands.txt
echo exit >> commands.txt
diskpart /s commands.txt > output.txt
type output.txt
del commands.txt
del output.txt
pause

@echo off
%1 mshta vbscript:CreateObject("Shell.Application").ShellExecute("cmd.exe","/c %~s0 ::","","runas",1)(window.close)&&exit
cd /d "%~dp0"
rem 上面是bat批处理以管理员权限运行
rem 下面是隐藏【卷0】【R盘】的命令
echo list volume > script.txt
echo select volume=0 >> script.txt
echo remove letter=R >> script.txt
DiskPart /s script.txt
del script.txt
rem 添加提示
echo 操作已完成，请按任意键继续...
pause >nul
